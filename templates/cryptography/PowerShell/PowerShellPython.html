import os
import time
import platform
import subprocess

def cls() -> None:
    os.system('')
    try:
        os.system(
            {'Windows': 'cls', 'Linux': 'clear'}[platform.system()]
        )
    except:
        print('\033[H\033[J', end='')
cls()

def PowerShell() -> None:
    cmd = f'powershell -Command Get-FileHash -Path {__file__} -Algorithm md5'
    startTime = time.time()
    process = subprocess.Popen(cmd, shell=True)
    process.wait()
    endTime = time.time()
    elapsedTime = endTime - startTime
    print(f"PowerShell Cost: {elapsedTime:.3f} s")


if __name__ == '__main__':
    PowerShell()
